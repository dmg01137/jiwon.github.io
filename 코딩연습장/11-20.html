1012. 실수 1개 입력받아 그대로 출력하기
#include <iostream>
            using namespace std;
            
            int main(){
                float f;
                cin >> f;
                cout << fixed; // 소수점 이하 고정
                cout.precision(6); 
                // fixed와 함께 쓰일 때 소수점 이하 개수를 6으로 설정
                cout << f;
                return 0;
            }

1019. 연월일 입력받아 그대로 출력하기

            using namespace std;
            
            int main(){
                int y, m, d;

                return 0;
            }
